168 while (true) {
169 unsigned k = leftChild(i);
170 if (k >= H.size())
171 break;
172 if (k+1 < H.size() && *H[k+1] < *H[k])
173 ++k; // right child of i
174 if ( ! (*H[k] < *x) )
175 break;
176 set(i, H[k]);
177 i = k;
178 }
