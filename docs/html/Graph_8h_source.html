394 adj.push_back(newEdge);
395 dest->incoming.push_back(newEdge);

399/*
400 * Auxiliary function to remove an outgoing edge (with a given destination (d))
401 * from a vertex (this).
402 * Returns true if successful, and false if such edge does not exist.
403 */
404template <class T>
405bool Vertex<T>::removeEdge(T in) {
406 bool removedEdge = false;
407 auto it = adj.begin();
408 while (it != adj.end()) {
409 Edge<T> *edge = *it;
410 Vertex<T> *dest = edge->getDest();
411 if (dest->getInfo() == in) {
412 it = adj.erase(it);
413 deleteEdge(edge);
414 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
415 }
416 else {
417 it++;
418 }
419 }
420 return removedEdge;
421}

423/*
424 * Auxiliary function to remove an outgoing edge of a vertex.
425 */
426template <class T>
427void Vertex<T>::removeOutgoingEdges() {
428 auto it = adj.begin();
429 while (it != adj.end()) {
430 Edge<T> *edge = *it;
431 it = adj.erase(it);
432 deleteEdge(edge);
433 }
434}

617/*
618 * Removes a vertex with a given content (in) from a graph (this), and
619 * all outgoing and incoming edges.
620 * Returns true if successful, and false if such vertex does not exist.
621 */
622template <class T>
623bool Graph<T>::removeVertex(const T &in) {
624 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
625 if ((*it)->getInfo() == in) {
626 auto v = *it;
627 v->removeOutgoingEdges();
628 for (auto u : vertexSet) {
629 u->removeEdge(v->getInfo());
630 }
631 vertexSet.erase(it);
632 delete v;
633 return true;
634 }
635 }
636 return false;
637}

668template <class T>
669bool Graph<T>::addBidirectionalEdge(const T &sourc, const T &dest, double d, double w) {
670 auto v1 = findVertex(sourc);
671 auto v2 = findVertex(dest);
672 if (v1 == nullptr || v2 == nullptr)
673 return false;
674 auto e1 = v1->addEdge(v2, d, w);
675 auto e2 = v2->addEdge(v1, d, w);
676 e1->setReverse(e2);
677 e2->setReverse(e1);
678 return true;

698template<class T>
699void Graph<T>::avoidEdges(std::vector<std::pair<int,int>> edges) {
700 if (!edges.empty()) {
701 for (auto p : edges) {
702 int sId = p.first;
703 int dId = p.second;
704
705 Vertex<T>* source = findLocationId(sId);
706 Vertex<T>* dest = findLocationId(dId);
707
708 if(source!=nullptr && dest!=nullptr) source->removeEdge(dest->getInfo());
709 }
710 }

715Graph<T>* copyGraph(Graph<T>* g) {
716 Graph<T>* gC = new Graph<T>();
717
718 // we add the vertices
719 for (auto v : g->getVertexSet()) {
720 gC->addVertex(v->getInfo());
721 }
722
723 // and build the edges
724 for (auto v : g->getVertexSet()) {

727 }
